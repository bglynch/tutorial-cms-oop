classmethods and staticmethods - 15mins
3 types of methods; Regular, Class and Static
Regular - automatically take the instance as the first argument (ie self)
Class.   - automatically take the class as the first argument (ie cls), need to use the class method decorator ( @classmethod)
Static.   - dont pass anything automatically, are included because they have a logical connection with the Class

# Pyhton Object-Oriented Programming
'''
This file shows how to use Static Methods

'''
# Create a class
class Employee:
    
    # Class Variables
    num_of_emps = 0
    raise_amount = 1.04
    
    def __init__(self, first, last, pay):
    # Instance Variables
        self.first = first
        self.last = last
        self.pay = pay
        self.email = first+'.'+last+'@company.com'
        
        Employee.num_of_emps += 1
        
    # Regular Methods
    def fullname(self):
        return '{} {}'.format(self.first, self.last)
    
    def apply_raise(self):
        self.pay = int(self.pay * self.raise_amount)
    
    # Class Methods
    @classmethod
    def set_raise_amt(cls, amount):
        cls.raise_amount = amount
    
    # Class Method as an alt constructor
    @classmethod
    def from_string(cls, emp_str):
        first, last, pay = emp_str.split('-')
        return cls(first, last, pay)
    
    # Static Method
    @staticmethod
    def is_workday(day):
        if day.weekday() == 5 or day.weekday() == 6:
            return False
        return True

            self.employees.remove(emp)

# Pyhton Object-Oriented Programming
'''
This file shows Special Methods we can use in Classes
Thses are also know as Magic Methods
'''
# Create a class
class Employee:
    
    # Class Variables
    raise_amount = 1.04
    
    def __init__(self, first, last, pay):
    # Instance Variables
        self.first = first
        self.last = last
        self.pay = pay
        self.email = first+'.'+last+'@company.com'
        
        
    # Regular Methods
    def fullname(self):
        return '{} {}'.format(self.first, self.last)
    
    def apply_raise(self):
        self.pay = int(self.pay * self.raise_amount)
        
    def __repr__(self):
        return "Employee('{}', '{}', '{}')".format(self.first, self.last, self.pay)
        
    def __str__(self):
        return "{} - {}".format(self.fullname(), self.email)
        
    def __add__(self, other):
        return self.pay + other.pay


# Instance of the Employee Class

Can use a deleter decorator to delete values
'''
This file shows the use of Property Decorators
'''
class Employee:
    
    def __init__(self, first, last):
        self.first = first
        self.last = last
    
    @property
    def email(self):
        return '{}.{}@email.com'.format(self.first, self.last)
        
    @property
    def fullname(self):
        return '{} {}'.format(self.first, self.last)
        
    @fullname.setter
    def fullname(self, name):
        first, last = name.split(' ')
        self.first = first
        self.last = last
        
    @fullname.deleter
    def fullname(self, name):
        print('Delete Name!')
        self.first = None
        self.last = None
    
emp_1 = Employee('Average','Joe')

emp_1.fullname = 'Brian Lynch'

print(emp_1.first)
print(emp_1.email)
print(emp_1.fullname)
Brian
Brian.Lynch@email.com
Brian Lynch

del emp_1.fullname
Delete Name!
